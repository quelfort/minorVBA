

Sub Ajeitar()
Dim numero As Single
Dim posicao1 As Single
Dim posicao2 As Variant
' Observações:
' 1. A função InStr é usada para procurar a virgula para a partir
'    dela podermos localizarmos nossa  variável
' 2. A função Mid é usada para pegar a varíavel entre virgulas

posicao2 = InStr(1, Range("A1"), ",")

posicao1 = 1
Contador! = 1

' Imprime o primeiro indivíduo

Cells(5, 5) = Mid(Range("A1"), 1, posicao2 - posicao1)
posicao1 = posicao2
Do While posicao2 <> 0

'Imprime o restante dos indivíduos

posicao2 = InStr((posicao1 + 1), Range("A1"), ",")
Cells(5, 5 + Contador) = Mid(Range("A1"), posicao1 + 1, posicao2 - posicao1 - 1)
posicao1 = posicao2

Contador = Contador + 1

Loop


posicao2 = InStr((posicao1 + 1), Range("A1"), ",")
' Imprime o ultimo indivíduo

Cells(5, 5 + Contador) = Mid(Range("A1"), posicao1 + 1, posicao2 - posicao1 - 1)





End Sub
